49 int lps331ap_init(lps331ap_t *dev, i2c_t i2c, uint8_t address, lps331ap_rate_t rate)
50 {
51  uint8_t tmp;
52 
53  /* save device specifics */
54  dev->i2c = i2c;
55  dev->address = address;
56 
57  /* Acquire exclusive access to the bus. */
58  i2c_acquire(dev->i2c);
59  /* initialize underlying I2C bus */
60  if (i2c_init_master(dev->i2c, BUS_SPEED) < 0) {
61  /* Release the bus for other threads. */
62  i2c_release(dev->i2c);
63  return -1;
64  }
65 
66  /* configure device, for simple operation only CTRL_REG1 needs to be touched */
67  tmp = LPS331AP_CTRL_REG1_DBDU | LPS331AP_CTRL_REG1_PD |
68  (rate << LPS331AP_CTRL_REG1_ODR_POS);
69  if (i2c_write_reg(dev->i2c, dev->address, LPS331AP_REG_CTRL_REG1, tmp) != 1) {
70  i2c_release(dev->i2c);
71  return -1;
72  }
73  i2c_release(dev->i2c);
74 
75  return 0;
76 }

99 {

103 
104  i2c_acquire(dev->i2c);
105  i2c_read_reg(dev->i2c, dev->address, LPS331AP_REG_PRESS_OUT_XL, &tmp);
106  val |= tmp;
107  i2c_read_reg(dev->i2c, dev->address, LPS331AP_REG_PRESS_OUT_L, &tmp);
108  val |= (tmp << 8);
109  i2c_read_reg(dev->i2c, dev->address, LPS331AP_REG_PRESS_OUT_H, &tmp);
110  i2c_release(dev->i2c);
111  val |= (tmp << 16);
112  /* see if value is negative */
113  if (tmp & 0x80) {
114  val |= (0xff << 24);
115  }
116 
117  /* compute actual pressure value in mbar */
118  res = ((float)val) / PRES_DIVIDER;
119 
120  return (int)res;
121 }
