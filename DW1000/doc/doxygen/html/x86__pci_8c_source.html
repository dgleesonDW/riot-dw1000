110 static void pci_setup_ios(struct x86_known_pci_device *dev)
111 {
112  /* § 6.2.5. (pp. 224) */
113 
114  unsigned bar_count = 0;
115  unsigned bar_base;
116 
117  unsigned header_type = x86_pci_read_reg(0x0c, dev->bus, dev->dev, dev->fun).header_type & PCI_HEADER_TYPE_MASK;
118  switch (header_type) {
119  case PCI_HEADER_TYPE_GENERAL_DEVICE:
120  bar_count = 6;
121  bar_base = 0x10;
122  break;
123  case PCI_HEADER_TYPE_BRIDGE:
124  bar_count = 2;
125  bar_base = 0x10;
126  break;
127  default:
128  printf(" Cannot configure header_type == 0x%02x, yet.\n", header_type);
129  return;
130  }
131 
132  for (unsigned bar_num = 0; bar_num < bar_count; ++bar_num) {
133  uint32_t old_bar = x86_pci_read(dev->bus, dev->dev, dev->fun, bar_base + 4 * bar_num);
134  if (old_bar == 0) {
135  continue;
136  }
137 
138  x86_pci_write(dev->bus, dev->dev, dev->fun, bar_base + 4 * bar_num, -1ul);
139  uint32_t tmp_bar = x86_pci_read(dev->bus, dev->dev, dev->fun, bar_base + 4 * bar_num);
140  x86_pci_write(dev->bus, dev->dev, dev->fun, bar_base + 4 * bar_num, old_bar);
141  if ((old_bar & PCI_BAR_IO_SPACE) != (tmp_bar & PCI_BAR_IO_SPACE)) {
142  /* cannot happen (?) */
143  continue;
144  }
145 
146  dev->io = realloc(dev->io, sizeof (*dev->io) * (dev->io_count + 1));
147  struct x86_pci_io *io = calloc(1, sizeof *io);
148  dev->io[dev->io_count] = io;
149  io->bar_num = bar_num;
150  ++dev->io_count;
151 
152  unsigned addr_offs = (tmp_bar & PCI_BAR_IO_SPACE) ? PCI_BAR_ADDR_OFFS_IO : PCI_BAR_ADDR_OFFS_MEM;
153  uint32_t length_tmp = tmp_bar >> addr_offs;
154  uint32_t length = 1 << addr_offs;
155  while ((length_tmp & 1) == 0) {
156  length <<= 1;
157  length_tmp >>= 1;
158  }
159  io->length = length;
160 
161  if (tmp_bar & PCI_BAR_IO_SPACE) {
162  io->type = PCI_IO_PORT;
163  io->addr.port = old_bar & ~((1 << PCI_BAR_ADDR_OFFS_IO) - 1);
164  printf(" BAR %u: I/O space, ports 0x%04x-0x%04x\n",
165  bar_num, io->addr.port, io->addr.port + length - 1);
166  }
167  else if ((old_bar & PCI_BAR_IO_SPACE) != PCI_BAR_SPACE_32 && (old_bar & PCI_BAR_IO_SPACE) != PCI_BAR_SPACE_64) {
168  printf(" BAR %u: memory with unknown location 0x%x, ERROR!\n", bar_num, (old_bar >> 1) & 3);
169  }
170  else {
171  uint32_t physical_start = old_bar & ~0xfff;
172  void *ptr = x86_map_physical_pages(physical_start, (length + 0xfff) / 0x1000, PT_P | PT_G | PT_RW | PT_PWT | PT_PCD | PT_XD);
173  if (!ptr) {
174  io->type = PCI_IO_INVALID;
175  printf(" BAR %u: memory, physical = 0x%08x-0x%08x, ERROR!\n",
176  bar_num, (unsigned) physical_start, physical_start + length - 1);
177  }
178  else {
179  io->type = PCI_IO_MEM;
180  io->addr.ptr = (char *) ptr + (old_bar & ~0xfff & ~((1 << PCI_BAR_ADDR_OFFS_MEM) - 1));
181  printf(" BAR %u: memory, physical = 0x%08x-0x%08x, virtual = 0x%08x-0x%08x\n",
182  bar_num,
183  physical_start, physical_start + length - 1,
184  (unsigned) ptr, (uintptr_t) ptr + length - 1);
185  }
186  }
187  }
188 }

192 void x86_pci_set_irq(struct x86_known_pci_device *d, uint8_t irq_num)
193 {
194  if (d->irq == irq_num) {
195  return;
196  }
197 
198  d->irq = irq_num;
199  uint32_t old_3c = x86_pci_read(d->bus, d->dev, d->fun, 0x3c);
200  x86_pci_write(d->bus, d->dev, d->fun, 0x3c, (old_3c & ~0xff) | d->irq);
201 
202  printf(" IRQ: new = %u, old = %u\n", d->irq, old_3c & 0xff);
203 }

205 static void pci_find_function(unsigned bus, unsigned dev, unsigned fun)
206 {
207  union pci_reg_0x00 vendor = x86_pci_read_reg(0x00, bus, dev, fun);
208  if (!pci_vendor_id_valid(vendor.vendor_id)) {
209  return;
210  }
211 
212  union pci_reg_0x08 class = x86_pci_read_reg(0x08, bus, dev, fun);
213  const char *baseclass_name, *subclass_name = x86_pci_subclass_to_string(class.baseclass,
214  class.subclass,
215  class.programming_interface,
216  &baseclass_name);
217  const char *vendor_name, *device_name = x86_pci_device_id_to_string(vendor.vendor_id, vendor.device_id, &vendor_name);
218  printf(" %02x:%02x.%x \"%s\": \"%s\" (%s: %s, rev: %02hhx)\n",
219  bus, dev, fun, vendor_name, device_name, baseclass_name, subclass_name, class.revision_id);
220 
221  /* cppcheck-suppress memleakOnRealloc */
222  known_pci_devices = realloc(known_pci_devices, sizeof (*known_pci_devices) * (num_known_pci_devices + 1));
223  struct x86_known_pci_device *d = calloc(1, sizeof *d);
224  known_pci_devices[num_known_pci_devices] = d;
225  ++num_known_pci_devices;
226 
227  d->bus = bus;
228  d->dev = dev;
229  d->fun = fun;
230  d->vendor = vendor;
231  d->class = class;
232  d->managed = false;
233 
234  uint32_t old_3c = x86_pci_read(bus, dev, fun, 0x3c);
235  if (old_3c & 0xff) {
236  d->irq = PCI_IRQ_DEFAULT;
237  x86_pci_write(bus, dev, fun, 0x3c, (old_3c & ~0xff) | d->irq);
238  printf(" IRQ: new = %u, old = %u\n", d->irq, old_3c & 0xff);
239  }
240 
241  pci_setup_ios(d);
242 
243  if (class.baseclass == 0x06 && class.subclass == 0x04) {
244  unsigned secondary_bus = pci_init_secondary_bus(bus, dev, fun);
245  if (secondary_bus != 0) {
246  pci_find_on_bus(secondary_bus);
247  }
248  }
249 }

251 static void pci_find_on_bus(unsigned bus)
252 {
253  for (unsigned dev = 0; dev < PCI_DEV_COUNT; ++dev) {
254  if (!pci_vendor_id_valid(x86_pci_read_reg(0x00, bus, dev, 0).vendor_id)) {
255  continue;
256  }
257 
258  if (x86_pci_read_reg(0x0c, bus, dev, 0).header_type & PCI_HEADER_TYPE_MULTI_FUNCTION) {
259  for (unsigned fun = 0; fun < PCI_FUN_COUNT; ++fun) {
260  pci_find_function(bus, dev, fun);
261  }
262  }
263  else {
264  pci_find_function(bus, dev, 0);
265  }
266  }
267 }

269 static void pci_find(void)
270 {
271  if (x86_pci_read_reg(0x0c, 0, 0, 0).header_type & PCI_HEADER_TYPE_MULTI_FUNCTION) {
272  for (unsigned fun = 0; fun < PCI_FUN_COUNT; ++fun) {
273  if (pci_vendor_id_valid(x86_pci_read_reg(0x00, 0, 0, fun).vendor_id)) {
274  pci_find_on_bus(fun);
275  }
276  }
277  }
278  else {
279  pci_find_on_bus(0);
280  }
