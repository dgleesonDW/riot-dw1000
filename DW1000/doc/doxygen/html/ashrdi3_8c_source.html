37 /*
38  * Shift a (signed) quad value right (arithmetic shift right).
39  */
40 quad_t __ashrdi3(quad_t a, qshift_t shift)
41 {
42  union uu aa;
43 
44  if (shift == 0) {
45  return a;
46  }
47 
48  aa.q = a;
49 
50  if (shift >= INT_BITS) {
51  /*
52  * Smear bits rightward using the machine's right-shift
53  * method, whether that is sign extension or zero fill,
54  * to get the `sign word' s. Note that shifting by
55  * INT_BITS is undefined, so we shift (INT_BITS-1),
56  * then 1 more, to get our answer.
57  */
58  /* LINTED inherits machine dependency */
59  int s = (aa.sl[H] >> (INT_BITS - 1)) >> 1;
60  /* LINTED inherits machine dependency*/
61  aa.ul[L] = aa.sl[H] >> (shift - INT_BITS);
62  aa.ul[H] = s;
63  }
64  else {
65  aa.ul[L] = (aa.ul[L] >> shift) | (aa.ul[H] << (INT_BITS - shift));
66  /* LINTED inherits machine dependency */
67  aa.sl[H] >>= shift;
68  }
69 
70  return aa.q;
71 }
