170 void mrf24j40_set_chan(mrf24j40_t *dev, uint8_t channel)
171 {
172  uint8_t channel_value;
173 
174  if ((channel < IEEE802154_CHANNEL_MIN) ||
175  (channel > IEEE802154_CHANNEL_MAX)) {
176  return;
177  }
178 
179  dev->netdev.chan = channel;
180 
181  /* Channel settings
182  * 11 -> Value = 0x03
183  * 12 -> Value = 0x13
184  * 13 -> Value = 0x23
185  * 14 -> Value = 0x33
186  * 15 -> Value = 0x43
187  * 16 -> Value = 0x53
188  * 17 -> Value = 0x63
189  * 18 -> Value = 0x73
190  * 19 -> Value = 0x83
191  * 20 -> Value = 0x93
192  * 21 -> Value = 0xA3
193  * 22 -> Value = 0xB3
194  * 23 -> Value = 0xC3
195  * 24 -> Value = 0xD3
196  * 25 -> Value = 0xE3
197  * 26 -> Value = 0xF3
198  */
199  /* not using an array here because it's not starting at zero */
200  switch (channel) {
201  case 11: channel_value = 0x03;
202  break;
203  case 12: channel_value = 0x13;
204  break;
205  case 13: channel_value = 0x23;
206  break;
207  case 14: channel_value = 0x33;
208  break;
209  case 15: channel_value = 0x43;
210  break;
211  case 16: channel_value = 0x53;
212  break;
213  case 17: channel_value = 0x63;
214  break;
215  case 18: channel_value = 0x73;
216  break;
217  case 19: channel_value = 0x83;
218  break;
219  case 20: channel_value = 0x93;
220  break;
221  case 21: channel_value = 0xa3;
222  break;
223  case 22: channel_value = 0xb3;
224  break;
225  case 23: channel_value = 0xc3;
226  break;
227  case 24: channel_value = 0xd3;
228  break;
229  case 25: channel_value = 0xe3;
230  break;
231  case 26: channel_value = 0xf3;
232  break;
233  default:
234  return;
235  }

237  mrf24j40_reg_write_long(dev, MRF24J40_REG_RFCON0, channel_value);

326 void mrf24j40_set_option(mrf24j40_t *dev, uint16_t option, bool state)
327 {
328  uint8_t tmp;
329 
330  DEBUG("set option %i to %i\n", option, state);
331 
332  /* set option field */
333  if (state) {
334  dev->netdev.flags |= option;
335  /* trigger option specific actions */
336  switch (option) {
337  case MRF24J40_OPT_CSMA:
338  DEBUG("[mrf24j40] opt: enabling CSMA mode (4 retries, macMinBE: 3 max BE: 5)\n");
339  /* Initialize CSMA seed with hardware address */
340  tmp = mrf24j40_reg_read_short(dev, MRF24J40_REG_TXMCR);
341  tmp &= ~MRF24J40_TXMCR_NOCSMA;
342  mrf24j40_reg_write_short(dev, MRF24J40_REG_TXMCR, tmp);
343  break;
344  case MRF24J40_OPT_PROMISCUOUS:
345  DEBUG("[mrf24j40] opt: enabling PROMISCUOUS mode\n");
346  /* disable auto ACKs in promiscuous mode */
347  tmp = mrf24j40_reg_read_short(dev, MRF24J40_REG_RXMCR);
348  tmp |= MRF24J40_RXMCR_NOACKRSP;
349  /* enable promiscuous mode */
350  tmp |= MRF24J40_RXMCR_PROMI;
351  tmp &= ~MRF24J40_RXMCR_ERRPKT;
352  mrf24j40_reg_write_short(dev, MRF24J40_REG_RXMCR, tmp);
353  break;
354  case NETDEV2_IEEE802154_ACK_REQ:
355  DEBUG("[mrf24j40] opt: enabling auto ACKs\n");
356  tmp = mrf24j40_reg_read_short(dev, MRF24J40_REG_RXMCR);
357  tmp &= ~MRF24J40_RXMCR_NOACKRSP;
358  mrf24j40_reg_write_short(dev, MRF24J40_REG_RXMCR, tmp);
359  break;
360  default:
361  /* do nothing */
362  break;
363  }
364  }
365  /* clear option field */
366  else {
367  dev->netdev.flags &= ~(option);
368  /* trigger option specific actions */
369  switch (option) {
370  case MRF24J40_OPT_CSMA:
371  DEBUG("[mrf24j40] opt: disabling CSMA mode\n");
372  tmp = mrf24j40_reg_read_short(dev, MRF24J40_REG_TXMCR);
373  tmp |= MRF24J40_TXMCR_NOCSMA;
374  /* MACMINBE<1:0>: The minimum value of the backoff exponent
375  * in the CSMA-CA algorithm. Note that if this value is set
376  * to ‘0’, collision avoidance is disabled. */
377  mrf24j40_reg_write_short(dev, MRF24J40_REG_TXMCR, tmp);
378  break;
379  case MRF24J40_OPT_PROMISCUOUS:
380  DEBUG("[mrf24j40] opt: disabling PROMISCUOUS mode\n");
381  /* disable promiscuous mode */
382  tmp = mrf24j40_reg_read_short(dev, MRF24J40_REG_RXMCR);
383  tmp &= ~MRF24J40_RXMCR_PROMI;
384  tmp &= ~MRF24J40_RXMCR_ERRPKT;
385  /* re-enable AUTOACK only if the option is set */
386  if (dev->netdev.flags & NETDEV2_IEEE802154_ACK_REQ) {
387  tmp &= ~(MRF24J40_RXMCR_NOACKRSP);
388  mrf24j40_reg_write_short(dev, MRF24J40_REG_RXMCR, tmp);
389  }
390  break;
391  case NETDEV2_IEEE802154_ACK_REQ:
392  DEBUG("[mrf24j40] opt: disabling auto ACKs\n");
393  tmp = mrf24j40_reg_read_short(dev, MRF24J40_REG_RXMCR);
394  tmp |= MRF24J40_RXMCR_NOACKRSP;
395  mrf24j40_reg_write_short(dev, MRF24J40_REG_RXMCR, tmp);
396  break;
397  default:
398  /* do nothing */
399  break;
400  }
401  }
402 }

405 void mrf24j40_set_state(mrf24j40_t *dev, uint8_t state)
406 {
407  uint8_t old_state;
408 
409  old_state = dev->state;
410 
411 
412  if (state == old_state) {
413  return;
414  }
415  /* check if asked to wake up from sleep mode */
416  if (old_state == MRF24J40_PSEUDO_STATE_SLEEP) {
417  mrf24j40_assert_awake(dev);
418  }
419  if (state == MRF24J40_PSEUDO_STATE_SLEEP) {
420  mrf24j40_sleep(dev);
421  }
422  if (state == MRF24J40_PSEUDO_STATE_IDLE) {
423  dev->state = state;
424  }
425  dev->idle_state = state;
426 }

428 void mrf24j40_sleep(mrf24j40_t *dev)
429 {
430  DEBUG("[mrf24j40] Putting into sleep mode\n");
431  /* Datasheet chapter 3.15.2 IMMEDIATE SLEEP AND WAKE-UP MODE */
432  /* First force a Power Management Reset */
433  mrf24j40_reg_write_short(dev, MRF24J40_REG_SOFTRST, MRF24J40_SOFTRST_RSTPWR);
434  /* Go to SLEEP mode */
435  mrf24j40_reg_write_short(dev, MRF24J40_REG_SLPACK, MRF24J40_SLPACK_SLPACK);
436  dev->state = MRF24J40_PSEUDO_STATE_SLEEP;
437 }

446 void mrf24j40_assert_awake(mrf24j40_t *dev)
447 {
448  if (dev->state == MRF24J40_PSEUDO_STATE_SLEEP) {
449  DEBUG("[mrf24j40] Waking up from sleep mode\n");
450  /* Wake mrf up */
451  mrf24j40_reg_write_short(dev, MRF24J40_REG_WAKECON, MRF24J40_WAKECON_IMMWAKE | MRF24J40_WAKECON_REGWAKE);
452  /* undocumented delay, needed for stable wakeup */
453  xtimer_usleep(MRF24J40_DELAY_SLEEP_TOGGLE);
454  mrf24j40_reg_write_short(dev, MRF24J40_REG_WAKECON, MRF24J40_WAKECON_IMMWAKE);
455  /* reset state machine */
456  mrf24j40_reg_write_short(dev, MRF24J40_REG_RFCTL, MRF24J40_RFCTL_RFRST);
457  mrf24j40_reg_write_short(dev, MRF24J40_REG_RFCTL, 0x00);
458  /* After wake-up, delay at least 2 ms to allow 20 MHz main
459  * oscillator time to stabilize before transmitting or receiving.
460  */
461  xtimer_usleep(MRF24J40_WAKEUP_DELAY);
462  /* reset interrupts */
463  mrf24j40_reg_read_short(dev, MRF24J40_REG_INTSTAT);
464  dev->state = MRF24J40_PSEUDO_STATE_IDLE;
465  }

468 void mrf24j40_reset_state_machine(mrf24j40_t *dev)
469 {
470  uint8_t rfstate;
471 
472  mrf24j40_reg_write_short(dev, MRF24J40_REG_RFCTL, MRF24J40_RFCTL_RFRST);
473  mrf24j40_reg_write_short(dev, MRF24J40_REG_RFCTL, 0x00);
474  xtimer_usleep(MRF24J40_STATE_RESET_DELAY); /* Delay at least 192us */
475  do {
476  rfstate = mrf24j40_reg_read_long(dev, MRF24J40_REG_RFSTATE);
477  } while ((rfstate & MRF24J40_RFSTATE_MASK) != MRF24J40_RFSTATE_RX);
478 }
