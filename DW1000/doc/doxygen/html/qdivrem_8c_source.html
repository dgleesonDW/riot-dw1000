56 /*
57  * __qdivrem(u, v, rem) returns u/v and, optionally, sets *rem to u%v.
58  *
59  * We do this in base 2-sup-HALF_BITS, so that all intermediate products
60  * fit within u_int. As a consequence, the maximum length dividend and
61  * divisor are 4 `digits' in this base (they are shorter if they have
62  * leading zeros).
63  */
64 u_quad_t
65 __qdivrem(u_quad_t uq, u_quad_t vq, u_quad_t *arq)
66 {
67  union uu tmp;
68  digit *u, *v, *q;
69  digit v1, v2;
70  u_int qhat, rhat, t;
71  int m, n, d, j, i;
72  digit uspace[5], vspace[5], qspace[5];
73 
74  /*
75  * Take care of special cases: divide by zero, and u < v.
76  */
77  if (vq == 0) {
78  /* divide by zero. */
79  static volatile const unsigned int zero = 0;
80 
81  tmp.ul[H] = tmp.ul[L] = 1 / zero;
82 
83  if (arq) {
84  *arq = uq;
85  }
86 
87  return tmp.q;
88  }
89 
90  if (uq < vq) {
91  if (arq) {
92  *arq = uq;
93  }
94 
95  return 0;
96  }
97 
98  u = &uspace[0];
99  v = &vspace[0];
100  q = &qspace[0];
101 
102  /*
103  * Break dividend and divisor into digits in base B, then
104  * count leading zeros to determine m and n. When done, we
105  * will have:
106  * u = (u[1]u[2]...u[m+n]) sub B
107  * v = (v[1]v[2]...v[n]) sub B
108  * v[1] != 0
109  * 1 < n <= 4 (if n = 1, we use a different division algorithm)
110  * m >= 0 (otherwise u < v, which we already checked)
111  * m + n = 4
112  * and thus
113  * m = 4 - n <= 2
114  */
115  tmp.uq = uq;
116  u[0] = 0;
117  u[1] = (digit) HHALF(tmp.ul[H]);
118  u[2] = (digit) LHALF(tmp.ul[H]);
119  u[3] = (digit) HHALF(tmp.ul[L]);
120  u[4] = (digit) LHALF(tmp.ul[L]);
121  tmp.uq = vq;
122  v[1] = (digit) HHALF(tmp.ul[H]);
123  v[2] = (digit) LHALF(tmp.ul[H]);
124  v[3] = (digit) HHALF(tmp.ul[L]);
125  v[4] = (digit) LHALF(tmp.ul[L]);
126 
127  for (n = 4; v[1] == 0; v++) {
128  if (--n == 1) {
129  u_int rbj; /* r*B+u[j] (not root boy jim) */
130  digit q1, q2, q3, q4;
131 
132  /*
133  * Change of plan, per exercise 16.
134  * r = 0;
135  * for j = 1..4:
136  * q[j] = floor((r*B + u[j]) / v),
137  * r = (r*B + u[j]) % v;
138  * We unroll this completely here.
139  */
140  t = v[2]; /* nonzero, by definition */
141  q1 = (digit) (u[1] / t);
142  rbj = COMBINE(u[1] % t, u[2]);
143  q2 = (digit) (rbj / t);
144  rbj = COMBINE(rbj % t, u[3]);
145  q3 = (digit) (rbj / t);
146  rbj = COMBINE(rbj % t, u[4]);
147  q4 = (digit) (rbj / t);
148 
149  if (arq) {
150  *arq = rbj % t;
151  }
152 
153  tmp.ul[H] = COMBINE(q1, q2);
154  tmp.ul[L] = COMBINE(q3, q4);
155  return tmp.q;
156  }
157  }
158 
159  /*
160  * By adjusting q once we determine m, we can guarantee that
161  * there is a complete four-digit quotient at &qspace[1] when
162  * we finally stop.
163  */
164  for (m = 4 - n; u[1] == 0; u++) {
165  m--;
166  }
167 
168  for (i = 4 - m; --i >= 0;) {
169  q[i] = 0;
170  }
171 
172  q += 4 - m;
173 
174  /*
175  * Here we run Program D, translated from MIX to C and acquiring
176  * a few minor changes.
177  *
178  * D1: choose multiplier 1 << d to ensure v[1] >= B/2.
179  */
180  d = 0;
181 
182  for (t = v[1]; t < B / 2; t <<= 1) {
183  d++;
184  }
185 
186  if (d > 0) {
187  shl(&u[0], m + n, d); /* u <<= d */
188  shl(&v[1], n - 1, d); /* v <<= d */
189  }
190 
191  /*
192  * D2: j = 0.
193  */
194  j = 0;
195  v1 = v[1]; /* for D3 -- note that v[1..n] are constant */
196  v2 = v[2]; /* for D3 */
197 
198  do {
199  digit uj0, uj1, uj2;
200 
201  /*
202  * D3: Calculate qhat (\^q, in TeX notation).
203  * Let qhat = min((u[j]*B + u[j+1])/v[1], B-1), and
204  * let rhat = (u[j]*B + u[j+1]) mod v[1].
205  * While rhat < B and v[2]*qhat > rhat*B+u[j+2],
206  * decrement qhat and increase rhat correspondingly.
207  * Note that if rhat >= B, v[2]*qhat < rhat*B.
208  */
209  uj0 = u[j + 0]; /* for D3 only -- note that u[j+...] change */
210  uj1 = u[j + 1]; /* for D3 only */
211  uj2 = u[j + 2]; /* for D3 only */
212 
213  if (uj0 == v1) {
214  qhat = B;
215  rhat = uj1;
216  goto qhat_too_big;
217  }
218  else {
219  u_int nn = COMBINE(uj0, uj1);
220  qhat = nn / v1;
221  rhat = nn % v1;
222  }
223 
224  while (v2 * qhat > COMBINE(rhat, uj2)) {
225 qhat_too_big:
226  qhat--;
227 
228  if ((rhat += v1) >= B) {
229  break;
230  }
231  }
232 
233  /*
234  * D4: Multiply and subtract.
235  * The variable `t' holds any borrows across the loop.
236  * We split this up so that we do not require v[0] = 0,
237  * and to eliminate a final special case.
238  */
239  for (t = 0, i = n; i > 0; i--) {
240  t = u[i + j] - v[i] * qhat - t;
241  u[i + j] = (digit) LHALF(t);
242  t = (B - HHALF(t)) & (B - 1);
243  }
244 
245  t = u[j] - t;
246  u[j] = (digit) LHALF(t);
247 
248  /*
249  * D5: test remainder.
250  * There is a borrow if and only if HHALF(t) is nonzero;
251  * in that (rare) case, qhat was too large (by exactly 1).
252  * Fix it by adding v[1..n] to u[j..j+n].
253  */
254  if (HHALF(t)) {
255  qhat--;
256 
257  for (t = 0, i = n; i > 0; i--) { /* D6: add back. */
258  t += u[i + j] + v[i];
259  u[i + j] = (digit) LHALF(t);
260  t = HHALF(t);
261  }
262 
263  u[j] = (digit) LHALF(u[j] + t);
264  }
265 
266  q[j] = (digit)qhat;
267  } while (++j <= m); /* D7: loop on j. */
268 
269  /*
270  * If caller wants the remainder, we have to calculate it as
271  * u[m..m+n] >> d (this is at most n digits and thus fits in
272  * u[m+1..m+n], but we may need more source digits).
273  */
274  if (arq) {
275  if (d) {
276  for (i = m + n; i > m; --i) {
277  u[i] = (digit)(((u_int)u[i] >> d) | LHALF((u_int)u[i - 1] << (HALF_BITS - d)));
278  }
279 
280  u[i] = 0;
281  }
282 
283  tmp.ul[H] = COMBINE(uspace[1], uspace[2]);
284  tmp.ul[L] = COMBINE(uspace[3], uspace[4]);
285  *arq = tmp.q;
286  }
287 
288  tmp.ul[H] = COMBINE(qspace[1], qspace[2]);
289  tmp.ul[L] = COMBINE(qspace[3], qspace[4]);
290  return tmp.q;
291 }

293 /*
294  * Shift p[0]..p[len] left `sh' bits, ignoring any bits that
295  * `fall out' the left (there never will be any such anyway).
296  * We may assume len >= 0. NOTE THAT THIS WRITES len+1 DIGITS.
297  */
298 static void shl(digit *p, int len, int sh)
299 {
300  int i;
301 
302  for (i = 0; i < len; i++) {
303  p[i] = (digit) (LHALF((u_int)p[i] << sh) | ((u_int)p[i + 1] >> (HALF_BITS - sh)));
304  }
305 
306  p[i] = (digit)(LHALF((u_int)p[i] << sh));
307 }
