114 void
115 stamptime(void)
116 {
117  static long startsecs = 0, startmsecs = 0;
118  long secs, msecs;
119  struct timeval tv;
120  time_t t;
121  char timec[20];
122 
123  gettimeofday(&tv, NULL) ;
124  msecs = tv.tv_usec / 1000;
125  secs = tv.tv_sec;
126 
127  if (startsecs) {
128  secs -= startsecs;
129  msecs -= startmsecs;
130 
131  if (msecs < 0) {
132  secs--;
133  msecs += 1000;
134  }
135 
136  fprintf(stderr, "%04li.%03li ", secs, msecs);
137  }
138  else {
139  startsecs = secs;
140  startmsecs = msecs;
141  t = time(NULL);
142  struct tm *tmp = localtime(&t);
143  strftime(timec, sizeof(timec), "%T", tmp);
144  // fprintf(stderr,"\n%s.%03lu ",timec,msecs);
145  fprintf(stderr, "\n%s ", timec);
146  }
147 }

149 int
150 is_sensible_string(const unsigned char *s, size_t len)
151 {
152  if(len > 0) {
153  if (s[0] == 0x60) {
154  /* Possibly IPv6 packet with default traffic class, assume non-printable */
155  return 0;
156  }
157  }
158 
159  int ret = 0;
160  for (size_t i = 0; i < len; i++) {
161  if (s[i] == '\r' || s[i] == '\n' || s[i] == '\t') {
162  continue;
163  }
164  else if (s[i] < ' ' || '~' < s[i]) {
165  return 0;
166  }
167  /* only return 1 if the string contains at least one letter or number */
168  if('A' <= s[i] && s[i] <= 'Z') {
169  ret = 1;
170  }
171  else if('a' <= s[i] && s[i] <= 'z') {
172  ret = 1;
173  }
174  else if('0' <= s[i] && s[i] <= '9') {
175  ret = 1;
176  }
177  }
178 
179  return ret;
180 }

182 /*
183  * Read from serial, when we have a packet write it to tun. No output
184  * buffering, input buffered by stdio.
185  */
186 void
187 serial_to_tun(FILE *inslip, int outfd)
188 {
189  static union {
190  unsigned char inbuf[2000];
191  } uip;
192  static unsigned int inbufptr = 0;
193  int ret;
194  unsigned char c;
195 
196 #ifdef __linux__
197  ret = fread(&c, 1, 1, inslip);
198 
199  if (ret == -1 || ret == 0) {
200  err(1, "serial_to_tun: read");
201  }
202 
203  goto after_fread;
204 #endif
205 
206  while (1) {
207  if (inbufptr >= sizeof(uip.inbuf)) {
208  if (timestamp) {
209  stamptime();
210  }
211 
212  fprintf(stderr, "*** dropping large %d byte packet\n", inbufptr);
213  inbufptr = 0;
214  }
215 
216  ret = fread(&c, 1, 1, inslip);
217 #ifdef __linux__
218  after_fread:
219 #endif
220 
221  if (ret == -1) {
222  err(1, "serial_to_tun: read");
223  }
224 
225  if (ret == 0) {
226  clearerr(inslip);
227  return;
228  }
229 
230  /* fprintf(stderr, ".");*/
231  switch (c) {
232  case SLIP_END:
233  if (inbufptr > 0) {
234  if (uip.inbuf[0] == '!') {
235  if (uip.inbuf[1] == 'M') {
236  /* Read gateway MAC address and autoconfigure tap0 interface */
237  char macs[24];
238  unsigned int pos = 0;
239 
240  for (unsigned int i = 0; i < 16; i++) {
241  macs[pos++] = uip.inbuf[2 + i];
242 
243  if ((i & 1) == 1 && i < 14) {
244  macs[pos++] = ':';
245  }
246  }
247 
248  if (timestamp) {
249  stamptime();
250  }
251 
252  macs[pos] = '\0';
253  fprintf(stderr, "*** Gateway's MAC address: %s\n", macs);
254 
255  if (timestamp) {
256  stamptime();
257  }
258 
259  ssystem("ifconfig %s down", tundev);
260 
261  if (timestamp) {
262  stamptime();
263  }
264 
265  ssystem("ifconfig %s hw ether %s", tundev, &macs[6]);
266 
267  if (timestamp) {
268  stamptime();
269  }
270 
271  ssystem("ifconfig %s up", tundev);
272  }
273  }
274  else if (uip.inbuf[0] == '?') {
275  if (uip.inbuf[1] == 'P') {
276  /* Prefix info requested */
277  struct in6_addr addr;
278  char *s = strchr(ipaddr, '/');
279 
280  if (s != NULL) {
281  *s = '\0';
282  }
283 
284  inet_pton(AF_INET6, ipaddr, &addr);
285 
286  if (timestamp) {
287  stamptime();
288  }
289 
290  fprintf(stderr, "*** Address:%s => %02x%02x:%02x%02x:%02x%02x:%02x%02x\n",
291  ipaddr,
292  addr.s6_addr[0], addr.s6_addr[1],
293  addr.s6_addr[2], addr.s6_addr[3],
294  addr.s6_addr[4], addr.s6_addr[5],
295  addr.s6_addr[6], addr.s6_addr[7]);
296  slip_send(slipfd, '!');
297  slip_send(slipfd, 'P');
298 
299  for (unsigned int i = 0; i < 8; i++) {
300  /* need to call the slip_send_char for stuffing */
301  slip_send_char(slipfd, addr.s6_addr[i]);
302  }
303 
304  slip_send(slipfd, SLIP_END);
305  }
306 
307 #define DEBUG_LINE_MARKER '\r'
308  }
309  else if (uip.inbuf[0] == DEBUG_LINE_MARKER) {
310  fwrite(uip.inbuf + 1, inbufptr - 1, 1, stdout);
311  }
312  else if (is_sensible_string(uip.inbuf, inbufptr)) {
313  if (verbose == 1) { /* strings already echoed below for verbose>1 */
314  if (timestamp) {
315  stamptime();
316  }
317 
318  fwrite(uip.inbuf, inbufptr, 1, stdout);
319  }
320  }
321  else {
322  if (verbose > 2) {
323  if (timestamp) {
324  stamptime();
325  }
326 
327  printf("Packet from SLIP of length %d - write TUN\n", inbufptr);
328 
329  if (verbose > 4) {
330 #if WIRESHARK_IMPORT_FORMAT
331  printf("0000");
332 
333  for (unsigned int i = 0; i < inbufptr; i++) {
334  printf(" %02x", uip.inbuf[i]);
335  }
336 
337 #else
338  printf(" ");
339 
340  for (unsigned int i = 0; i < inbufptr; i++) {
341  printf("%02x", uip.inbuf[i]);
342 
343  if ((i & 3) == 3) {
344  printf(" ");
345  }
346 
347  if ((i & 15) == 15) {
348  printf("\n ");
349  }
350  }
351 
352 #endif
353  printf("\n");
354  }
355  }
356 
357  if (write(outfd, uip.inbuf, inbufptr) != inbufptr) {
358  err(1, "serial_to_tun: write");
359  }
360  }
361 
362  inbufptr = 0;
363  }
364 
365  break;
366 
367  case SLIP_ESC:
368  if (fread(&c, 1, 1, inslip) != 1) {
369  clearerr(inslip);
370  /* Put ESC back and give up! */
371  ungetc(SLIP_ESC, inslip);
372  return;
373  }
374 
375  switch (c) {
376  case SLIP_ESC_END:
377  c = SLIP_END;
378  break;
379 
380  case SLIP_ESC_ESC:
381  c = SLIP_ESC;
382  break;
383  }
384 
385  /* FALLTHROUGH */
386  default:
387  uip.inbuf[inbufptr++] = c;
388 
389  /* Echo lines as they are received for verbose=2,3,5+ */
390  /* Echo all printable characters for verbose==4 */
391  if ((verbose == 2) || (verbose == 3) || (verbose > 4)) {
392  if (c == '\n') {
393  if (is_sensible_string(uip.inbuf, inbufptr)) {
394  if (timestamp) {
395  stamptime();
396  }
397 
398  fwrite(uip.inbuf, inbufptr, 1, stdout);
399  inbufptr = 0;
400  }
401  }
402  }
403  else if (verbose == 4) {
404  if (c == 0 || c == '\r' || c == '\n' || c == '\t' || (c >= ' ' && c <= '~')) {
405  fwrite(&c, 1, 1, stdout);
406 
407  if (c == '\n') if (timestamp) {
408  stamptime();
409  }
410  }
411  }
412 
413  break;
414  }
415  }
416 }

447 void
448 slip_flushbuf(int fd)
449 {
450  int n;
451 
452  if (slip_empty()) {
453  return;
454  }
455 
456  n = write(fd, slip_buf + slip_begin, (slip_end - slip_begin));
457 
458  if (n == -1 && errno != EAGAIN) {
459  err(1, "slip_flushbuf write failed");
460  }
461  else if (n == -1) {
462  PROGRESS("Q"); /* Outqueue is full! */
463  }
464  else {
465  slip_begin += n;
466 
467  if (slip_begin == slip_end) {
468  slip_begin = slip_end = 0;
469  }
470  }

582 void
583 stty_telos(int fd)
584 {
585  struct termios tty;
586  speed_t speed = b_rate;
587  int i;
588 
589  if (tcflush(fd, TCIOFLUSH) == -1) {
590  err(1, "tcflush");
591  }
592 
593  if (tcgetattr(fd, &tty) == -1) {
594  err(1, "tcgetattr");
595  }
596 
597  cfmakeraw(&tty);
598 
599  /* Nonblocking read. */
600  tty.c_cc[VTIME] = 0;
601  tty.c_cc[VMIN] = 0;
602 
603  if (flowcontrol) {
604  tty.c_cflag |= CRTSCTS;
605  }
606  else {
607  tty.c_cflag &= ~CRTSCTS;
608  }
609 
610  tty.c_cflag &= ~HUPCL;
611  tty.c_cflag &= ~CLOCAL;
612 
613  cfsetispeed(&tty, speed);
614  cfsetospeed(&tty, speed);
615 
616  if (tcsetattr(fd, TCSAFLUSH, &tty) == -1) {
617  err(1, "tcsetattr");
618  }
619 
620 #if 1
621  /* Nonblocking read and write. */
622  /* if (fcntl(fd, F_SETFL, O_NONBLOCK) == -1) err(1, "fcntl"); */
623 
624  tty.c_cflag |= CLOCAL;
625 
626  if (tcsetattr(fd, TCSAFLUSH, &tty) == -1) {
627  err(1, "tcsetattr");
628  }
629 
630  i = TIOCM_DTR;
631 
632  if (ioctl(fd, TIOCMBIS, &i) == -1) {
633  err(1, "ioctl");
634  }
635 
636 #endif
637 
638  usleep(10 * 1000); /* Wait for hardware 10ms. */
639 
640  /* Flush input and output buffers. */
641  if (tcflush(fd, TCIOFLUSH) == -1) {
642  err(1, "tcflush");
643  }
644 }

778 void
779 ifconf(const char *tundev, const char *ipaddr)
780 {
781 #ifdef __linux__
782 
783  if (timestamp) {
784  stamptime();
785  }
786 
787  ssystem("ifconfig %s inet `hostname` up", tundev);
788 
789  if (timestamp) {
790  stamptime();
791  }
792 
793  ssystem("ifconfig %s add %s", tundev, ipaddr);
794 
795  /* radvd needs a link local address for routing */
796 #if 0
797  /* fe80::1/64 is good enough */
798  ssystem("ifconfig %s add fe80::1/64", tundev);
799 #elif 1
800  /* Generate a link local address a la sixxs/aiccu */
801  /* First a full parse, stripping off the prefix length */
802  {
803  char lladdr[40];
804  char c, *ptr = (char *)ipaddr;
805  uint16_t digit, ai, a[8], cc, scc, i;
806 
807  for (ai = 0; ai < 8; ai++) {
808  a[ai] = 0;
809  }
